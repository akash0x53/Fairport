00974 inline size_t fairport::extended_block::read_raw(byte* pdest_buffer, size_t size, ulong offset) const
00975 {
00976     assert(offset <= get_total_size());
00977 
00978     if(offset + size > get_total_size())
00979         size = get_total_size() - offset;
00980 
00981     byte* pend = pdest_buffer + size;
00982 
00983     size_t total_bytes_read = 0;
00984 
00985     while(pdest_buffer != pend)
00986     {
00987         // the child this read starts on
00988         uint child_pos = offset / m_child_max_total_size;
00989         // offset into the child block this read starts on
00990         ulong child_offset = offset % m_child_max_total_size;
00991 
00992         // call into our child to read the data
00993         size_t bytes_read = get_child_block(child_pos)->read_raw(pdest_buffer, size, child_offset);
00994         assert(bytes_read <= size);
00995     
00996         // adjust pointers accordingly
00997         pdest_buffer += bytes_read;
00998         offset += bytes_read;
00999         size -= bytes_read;
01000         total_bytes_read += bytes_read;
01001 
01002         assert(pdest_buffer <= pend);
01003     }
01004 
01005     return total_bytes_read;
