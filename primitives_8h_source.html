00041 // Many of the low-level data structures in Fairport rely on being laid out
00042 // in the fashion preferred by Visual C++.  For example, these structures
00043 // need to align ulonglong to the nearest 8 bytes.  Under GCC, we can force
00044 // this layout behavior on Mac and Linux systems by appending an
00045 // appropriate attribute declaration to the structure.  We're also tried
00046 // '#pragma ms_struct on', but it fails on at least some Linux systems.
00047 #ifdef __GNUC__
00048 #define FAIRPORT_MS_STRUCT __attribute__((ms_struct))
00049 #define FAIRPORT_PACK_2 __attribute__((packed, aligned(2)))
