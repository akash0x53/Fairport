00048 // Many of the low-level data structures in Fairport rely on being laid out
00049 // in the fashion preferred by Visual C++.  For example, these structures
00050 // need to align ulonglong to the nearest 8 bytes.  Under GCC, we can force
00051 // this layout behavior on Mac and Linux systems by appending an
00052 // appropriate attribute declaration to the structure.  We're also tried
00053 // '#pragma ms_struct on', but it fails on at least some Linux systems.
00054 #ifdef __GNUC__
00055 #define FAIRPORT_MS_STRUCT __attribute__((ms_struct))
00056 #define FAIRPORT_PACK_2 __attribute__((packed, aligned(2)))
00057 #else
00058 #define FAIRPORT_MS_STRUCT
00059 #define FAIRPORT_PACK_2
00060 #endif
